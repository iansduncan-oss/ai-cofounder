Add a `redis` service to `docker-compose.prod.yml` matching the dev compose pattern, plus a `worker` service using the same agent-server image with a different CMD.

**Redis service** (add before `agent-server`):
```yaml
redis:
  image: redis:7-alpine
  container_name: ai-cofounder-redis
  restart: unless-stopped
  volumes:
    - redisdata:/data
  command: redis-server --appendonly yes --maxmemory 256mb --maxmemory-policy allkeys-lru
  healthcheck:
    test: ["CMD", "redis-cli", "ping"]
    interval: 5s
    timeout: 5s
    retries: 5
  networks:
    - avion_avion_net
```
Note: No ports mapping in prod (only accessible within Docker network). This is intentional for security.

**Worker service** (add after `agent-server`):
```yaml
worker:
  image: ai-cofounder-agent-server:latest
  container_name: ai-cofounder-worker
  restart: unless-stopped
  command: ["node", "apps/agent-server/dist/worker.js"]
  stop_grace_period: 120s
  environment:
    - DATABASE_URL=${DATABASE_URL}
    - REDIS_URL=redis://redis:6379
    - ANTHROPIC_API_KEY=${ANTHROPIC_API_KEY}
    - ANTHROPIC_MODEL=${ANTHROPIC_MODEL:-claude-sonnet-4-20250514}
    - GROQ_API_KEY=${GROQ_API_KEY:-}
    - GEMINI_API_KEY=${GEMINI_API_KEY:-}

    - TAVILY_API_KEY=${TAVILY_API_KEY:-}
    - WORKSPACE_DIR=/data/workspace
    - LOG_LEVEL=${LOG_LEVEL:-info}
  volumes:
    - workspace_data:/data/workspace
    - /var/run/docker.sock:/var/run/docker.sock
  networks:
    - avion_avion_net
  depends_on:
    redis:
      condition: service_healthy
```
