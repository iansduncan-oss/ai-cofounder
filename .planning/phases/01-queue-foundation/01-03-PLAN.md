**Step 1 — Add `getJobStatus()` to `packages/queue/src/helpers.ts`:**

Add a helper function at the bottom of the file that wraps BullMQ's `Job.fromId()` and `job.getState()`. This keeps the `bullmq` direct import inside the queue package rather than in route handlers.

```typescript
import { Job } from "bullmq";

export interface JobStatusResult {
  state: "waiting" | "active" | "completed" | "failed" | "delayed" | "unknown";
  jobId: string;
  attemptsMade: number;
  finishedOn?: number;
  failedReason?: string;
}

export async function getJobStatus(jobId: string): Promise<JobStatusResult | null> {
  const queue = getAgentTaskQueue();
  const job = await Job.fromId(queue, jobId);
  if (!job) return null;

  const state = await job.getState();
  return {
    state,
    jobId,
    attemptsMade: job.attemptsMade,
    finishedOn: job.finishedOn,
    failedReason: job.failedReason,
  };
}
```

Add a new route at the bottom of `goalRoutes`:

```typescript
import { getJobStatus } from "@ai-cofounder/queue";

/* GET /:id/queue-status — query BullMQ job state for this goal */
app.get<{ Params: typeof IdParams.static }>(
  "/:id/queue-status",
  { schema: { tags: ["goals"], params: IdParams } },
  async (request, reply) => {
    const goal = await getGoal(app.db, request.params.id);
    if (!goal) return reply.status(404).send({ error: "Goal not found" });

    const metadata = goal.metadata as Record<string, unknown> | null;
    const jobId = metadata?.queueJobId as string | undefined;

    if (!jobId) {
      return { status: "not_queued", goalStatus: goal.status };
    }

    const jobStatus = await getJobStatus(jobId);
    if (!jobStatus) {
      return { status: "not_found", jobId };
    }

    return {
      status: jobStatus.state,
      jobId,
      attemptsMade: jobStatus.attemptsMade,
      finishedOn: jobStatus.finishedOn,
      failedReason: jobStatus.failedReason,
    };
  },
);
```

Note: The `getJobStatus` import from `@ai-cofounder/queue` requires the package to be built first (Step 2 build step). If the linter strips the import before build, run `npm run build -w @ai-cofounder/queue` first.

**Step 1 — Add `pingRedis()` helper to `packages/queue/src/helpers.ts`:**

Add a Redis ping function that creates a short-lived ioredis connection to test connectivity. This is better than importing ioredis directly in the route.

```typescript
import Redis from "ioredis";
import { getRedisConnection } from "./connection.js";

export async function pingRedis(): Promise<"ok" | "unreachable"> {
  try {
    const connOpts = getRedisConnection();
    const redis = new Redis({
      host: connOpts.host,
      port: connOpts.port,
      password: connOpts.password,
      lazyConnect: true,
      connectTimeout: 3000,
      maxRetriesPerRequest: 0,
      enableReadyCheck: false,
    });
    await redis.connect();
    await redis.ping();
    await redis.quit();
    return "ok";
  } catch {
    return "unreachable";
  }
}
```

Note on ioredis: It's a transitive dependency of bullmq (bullmq depends on ioredis). The import `import Redis from "ioredis"` should resolve. If TypeScript complains about missing types, add `"ioredis": "*"` to `packages/queue/package.json` dependencies (not devDependencies — it's used in production for the health ping).

Export from `packages/queue/src/index.ts`:

**Step 2 — Update `apps/agent-server/src/routes/health.ts`:**

Add Redis status to the main `/health` endpoint. Import `pingRedis` from `@ai-cofounder/queue` and `optionalEnv` from `@ai-cofounder/shared`.

Update the `GET /health` handler:

```typescript
import { pingRedis } from "@ai-cofounder/queue";
import { optionalEnv } from "@ai-cofounder/shared";

// Inside the handler:
app.get("/health", { schema: { tags: ["health"] } }, async (_request, reply) => {
  let dbStatus = "ok";
  try {
    await app.db.execute(sql`SELECT 1`);
  } catch {
    dbStatus = "unreachable";
  }

  // Redis health check — only if REDIS_URL is configured
  let redisStatus = "disabled";
  const redisUrl = optionalEnv("REDIS_URL", "");
  if (redisUrl) {
    redisStatus = await pingRedis();
  }

  const isHealthy = dbStatus === "ok" && (redisStatus === "ok" || redisStatus === "disabled");

  if (!isHealthy) {
    reply.code(503);
  }

  return {
    status: isHealthy ? "ok" : "degraded",
    timestamp: new Date().toISOString(),
    uptime: process.uptime(),
    database: dbStatus,
    redis: redisStatus,
  };
});
```

Key behaviors:
- If `REDIS_URL` is not set, redis status is `"disabled"` and health is still "ok" (Redis is optional for dev).
- If `REDIS_URL` is set but Redis is unreachable, status becomes "degraded" (503).
- The existing DB check is preserved.
- Response now includes `database` and `redis` fields.
