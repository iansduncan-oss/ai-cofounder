**Create `apps/agent-server/src/services/autonomy-tier.ts`:**

```typescript
export type AutonomyTier = "green" | "yellow" | "red";

export class AutonomyTierService {
  private tiers: Map<string, { tier: AutonomyTier; timeoutMs: number }>;
  private loaded: boolean;
  private loadPromise: Promise<void> | null; // mutex for reload

  constructor(private db: Db) {}

  async load(): Promise<void> { /* load from listToolTierConfigs, populate Map */ }

  getTier(toolName: string): AutonomyTier {
    return this.tiers.get(toolName)?.tier ?? "green"; // default green
  }

  getTimeoutMs(toolName: string): number {
    return this.tiers.get(toolName)?.timeoutMs ?? 300_000;
  }

  getAllRed(): string[] {
    return [...this.tiers.entries()]
      .filter(([, v]) => v.tier === "red")
      .map(([k]) => k);
  }

  async reload(): Promise<void> {
    // Use loadPromise as mutex to prevent concurrent reloads
    if (this.loadPromise) return this.loadPromise;
    this.loadPromise = this.load();
    await this.loadPromise;
    this.loadPromise = null;
  }
}
```

2. Create `executeWithTierCheck(block, services, context)` function:
   - If no `autonomyTierService` on services, fall through to `executeSharedTool` (backward compat).
   - Get tier via `services.autonomyTierService.getTier(block.name)`.
   - **Red:** Return `{ error: "Tool X is in the red tier and cannot be executed. This operation has been blocked for safety." }`. Do NOT call executeSharedTool.
   - **Yellow:** Call `executeYellowTierTool(block, services, context)`:
     - Create approval via `createApproval(db, { taskId: context.goalId ?? undefined, requestedBy: (context.agentRole ?? "orchestrator") as AgentRole, reason: ... })`. Note taskId is now optional.
     - Call `notifyApprovalCreated(...)` to send Slack/Discord notification.
     - Poll DB every 2000ms until: approved (execute tool and return), rejected (return error), or timeout exceeded (auto-deny via resolveApproval, return error).
     - Timeout = `services.autonomyTierService.getTimeoutMs(block.name)`.
   - **Green:** Call `executeSharedTool` directly.
